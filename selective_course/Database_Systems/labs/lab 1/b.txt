select "Country_Region"
from (select "Country_Region", "Last_Update", sum("Confirmed")
	  from "March22"
	  group by "Country_Region", "Last_Update"
	  union
	  select "Country_Region", "Last_Update", sum("Confirmed")
	  from "March23"
	  group by "Country_Region", "Last_Update"
	 ) as tmp
where(sum > 1000000 and Date("Last_Update") = '2022-03-24')
